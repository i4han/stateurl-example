Type Alias SurlToFrom<Routes>

SurlToFrom<Routes>: PathsToRegistry<BuildPathsFromArray<Routes>>

Extract ToRegistry type from routes defined with defineRoutes(). Schema-aware: param slots use schema types when defined.

Schema param syntax:

  • 0 → any number
  • '' → any string
  • [10, 11, 12] → union 10|11|12 (first value is default)

Type Parameters

  • Routes extends readonly Route[]

Example

const routes = defineRoutes([
    { path: 'home', render: Home },
    { path: 'users', render: Users, outlet: [
        { path: 'profile/:userId', schema: { param: { userId: 0 } }, render: UserDetail }
    ]},
    { path: 'org/:orgId', schema: { param: { orgId: [10, 11, 12] } }, render: Org },
] as const)

type MySurlTo = SurlToFrom<typeof routes>
// Result: {
//   '/home': true
//   '/users': true
//   '/users/profile/${number}': true       // userId: 0 → any number
//   '/org/${'10' | '11' | '12'}': true     // orgId: [10,11,12] → constrained
// }

// Use in module augmentation:
declare module 'stateurl' {
    interface SurlTo extends SurlToFrom<typeof routes> {}
}

// Schema-aware validation:
to('/users/profile/:userId', { userId: 42 })  // ✓ named param
to('/users/profile/42')                        // ✓ hardcoded number
to('/org/10')                                  // ✓ valid orgId
to('/org/13')                                  // ✗ type error! 13 not in 10|11|12
